1. Find the Smaller Angle

 var Minimal_Angle = (h, m) => 
        {
         var hourangle=(h*30)+(0.5*m);
        var minangle=(m*6);
        var angle=Math.abs(hourangle-minangle);
        return(Math.min(360-angle,angle));
        };

3. Perfect Number Check.

 var Perfect_Check = (N) => 
            {
                  
                    let sum=0;
                    for(let i=1;i< =N;i++)
                    {
                      if(N%i===0){
                      sum=sum+i;
                      }
                    }
                    if(sum===N)
                    {
                      return "YES";
                    }
                    else
                    {
                      return "NO";
                    }
            };
