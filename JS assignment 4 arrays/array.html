9.Min and Max 

         function arrayMin(arr) 
 {
  let min=arr[0];
  N=arr.length;
  for(let i=0;i< N;i++)
  {
    if(arr[i+1]< min)
    {
      min=arr[i+1];
    }
  }
  return min;
 }

 function arrayMax(arr) 
 {
 let max=arr[0];
 N=arr.length;
 for(let i=0;i< N;i++)
 {
   if(arr[i+1]> max)
   {
     max=arr[i+1];
   }
 }
 return max;
 }
